1//! # Proxy Gatherer CLI
2//!
3//! This module provides the command-line interface (CLI) for the Gooty Proxy application.
4//! It allows users to manage configurations, test proxies, and scrape proxy sources.
5//!
6//! ## Overview
7//!
8//! The CLI supports the following commands:
9//! - `Config`: Manage configuration files (create or validate)
10//! - `Proxy`: Test and manage individual proxies
11//! - `Source`: Scrape proxies from websites and manage sources
12//!
13//! ## Examples
14//!
15//! ```
16//! // Run the CLI with the `Proxy` command to test a proxy
17//! gatherer proxy --judge "http://127.0.0.1:8080"
18//! ```
19

147/// Prints detailed information about a proxy to the console.
148///
149/// # Arguments
150/// * `proxy` - The proxy object containing information to display
151fn print_proxy_details(proxy: &Proxy) {
152    println!("Proxy Type: {}", proxy.proxy_type);
153    println!("Anonymity Level: {}", proxy.anonymity);
154    if let Some(latency) = proxy.latency_ms {
155        println!("Latency: {latency}ms");
156    }
157    if let Some(country) = &proxy.country {
158        println!("Country: {country}");
159    }
160    if let Some(org) = &proxy.organization {
161        println!("Organization: {org}");
162    }
163    if let Some(asn) = &proxy.asn {
164        println!("ASN: {asn}");
165    }
166    if let Some(hostname) = &proxy.hostname {
167        println!("Hostname: {hostname}");
168    }
169}

171/// Handles the Config command, creating or validating configuration files.
172///
173/// # Arguments
174/// * `create` - Optional path where to create default configuration
175/// * `validate` - Optional path to validate configuration
176///
177/// # Returns
178/// * `()` - The function exits the program with appropriate status code
179fn handle_config_command(create: Option<String>, validate: Option<String>) {
180    if let Some(path) = create {
181        // Create default configuration
182        let config = FilestoreConfig {
183            data_dir: path.clone(),
184            ..Default::default()
185        };
186        let filestore = match Filestore::with_config(config) {
187            Ok(fs) => fs,
188            Err(e) => {
189                eprintln!("Failed to create filestore: {e}");
190                std::process::exit(1);
191            }
192        };
193
194        // Create default configuration files
195        let default_config = AppConfig::default();
196        if let Err(e) = filestore.save_config(&default_config, "config") {
197            eprintln!("Failed to save configuration: {e}");
198            std::process::exit(1);
199        }
200
201        println!("Created default configuration in {path}");
202        std::process::exit(0);
203    }
204
205    if let Some(path) = validate {
206        // Validate existing configuration
207        let config = FilestoreConfig {
208            data_dir: path.clone(),
209            create_defaults_if_missing: false,
210            ..Default::default()
211        };
212        let filestore = match Filestore::with_config(config) {
213            Ok(fs) => fs,
214            Err(e) => {
215                eprintln!("Failed to access filestore: {e}");
216                std::process::exit(1);
217            }
218        };
219
220        // Try to load and validate configuration
221        match filestore.load_config("config") {
222            Ok(_) => {
223                println!("Configuration in {path} is valid");
224                std::process::exit(0);
225            }
226            Err(e) => {
227                eprintln!("Configuration validation failed: {e}");
228                std::process::exit(1);
229            }
230        }
231    }
232}

242    let mut manager = ProxyManager::new()?;
243
244    // Initialize judge

255/// Handles the Proxy command, testing individual proxies.
256///
257/// # Arguments
258/// * `judge` - Optional proxy URL to test
259/// * `dry` - Whether to avoid saving results
260///
261/// # Returns
262/// * `()` - The function exits the program with appropriate status code
263async fn handle_proxy_command(judge: Option<String>, dry: bool) {
264    if let Some(proxy_url) = judge {
265        // Initialize proxy manager and required components
266        let mut manager = match init_proxy_manager(true) {
267            Ok(m) => m,
268            Err(e) => {
269                eprintln!("Failed to initialize proxy manager: {e}");
270                std::process::exit(1);
271            }
272        };
273
274        // Parse proxy URL
275        let proxy = match parse_proxy_url(&proxy_url) {
276            Ok(p) => p,
277            Err(e) => {
278                eprintln!("Invalid proxy URL: {e}");
279                std::process::exit(1);
280            }
281        };
282
283        println!("Testing proxy: {proxy_url}");
284
285        // Add proxy to manager
286        if let Err(e) = manager.add_proxy(proxy) {
287            eprintln!("Failed to add proxy: {e}");
288            std::process::exit(1);
289        }
290
291        // Get proxy ID for management
292        let proxy_id = proxy_url.clone();
293
294        // Check proxy connectivity and anonymity
295        if let Err(e) = manager.check_proxy(&proxy_id).await {
296            eprintln!("Proxy test failed: {e}");
297            std::process::exit(1);
298        }
299
300        // Enrich with IP metadata
301        if let Err(e) = manager.enrich_proxy(&proxy_id).await {
302            eprintln!("Failed to enrich proxy data: {e}");
303            std::process::exit(1);
304        }
305
306        #[allow(clippy::cast_precision_loss)]
307        // Get the tested proxy
308        if let Some(proxy) = manager.get_proxy(&proxy_id) {
309            // Print detailed results
310            println!("\nProxy Test Results:");
311            println!("------------------");
312            print_proxy_details(proxy);
313
314            println!("\nTest Statistics:");
315            println!(
316                "Success Rate: {:.2}%",
317                (proxy.check_success_rate() as f64 / 100.0)
318            );
319            println!(
320                "Checks: {} total, {} failed",
321                proxy.check_count, proxy.check_failure_count
322            );
323
324            // Save to proxy list if test was successful and not in dry run mode
325            if !dry && proxy.check_success_rate() > 0 {
326                if let Some(filestore) = get_filestore("data") {
327                    match filestore.load_proxies("proxies") {
328                        Ok(mut proxies) => {
329                            proxies.push(proxy.clone());
330                            if let Err(e) = filestore.save_proxies(&proxies, "proxies") {
331                                eprintln!("Failed to save proxy: {e}");
332                            } else {
333                                println!("\nProxy saved to list successfully");
334                            }
335                        }
336                        Err(e) => eprintln!("Failed to load proxy list: {e}"),
337                    }
338                }
339            }
340        }
341
342        std::process::exit(0);
343    }
344}

360/// Tests and enriches proxies based on the specified judgement mode.
361///
362/// # Arguments
363/// * `proxies` - List of proxies to test
364/// * `mode` - Judgement mode determining the level of testing and enrichment
365///
366/// # Returns
367/// * `Result<Vec<Proxy>, Box<dyn std::error::Error>>` - The tested proxies or an error
368async fn test_and_enrich_proxies(
369    mut proxies: Vec<Proxy>,
370    mode: JudgementMode,
371) -> Result<Vec<Proxy>, Box<dyn std::error::Error>> {
372    if mode == JudgementMode::None {
373        return Ok(proxies);
374    }
375
376    // Initialize manager
377    let mut manager = init_proxy_manager(mode == JudgementMode::Full)?;
378
379    // Test proxies (basic connectivity)
380    println!("Testing proxies...");
381    let pb = ProgressBar::new(proxies.len() as u64);
382    pb.set_style(
383        ProgressStyle::default_bar()
384            .template("[{elapsed_precise}] {bar:40.cyan/blue} {pos:>7}/{len:7} {msg}")
385            .expect("Failed to create progress bar style")
386            .progress_chars("##-"),
387    );
388
389    // Check all proxies with progress
390    manager.check_all_proxies(&mut proxies, 10).await?;
391    pb.finish_with_message("Proxy testing complete");
392
393    // Gather additional information in full mode
394    if mode == JudgementMode::Full {
395        println!("\nGathering detailed proxy information...");
396        let pb = ProgressBar::new(proxies.len() as u64);
397        pb.set_style(
398            ProgressStyle::default_bar()
399                .template("[{elapsed_precise}] {bar:40.cyan/blue} {pos:>7}/{len:7} {msg}")
400                .expect("Failed to create progress bar style")
401                .progress_chars("##-"),
402        );
403
404        manager.enrich_all_proxies(&mut proxies, 10).await?;
405        pb.finish_with_message("Detail gathering complete");
406    }
407
408    Ok(proxies)
409}

411/// Handles saving proxies and sources to the filestore.
412///
413/// # Arguments
414/// * `proxies` - List of proxies to save
415/// * `source` - The source to save
416/// * `filestore` - Filestore for persistence
417/// * `dry` - Whether to avoid saving
418/// * `raw_response` - Optional raw response data to save
419/// * `mode` - Judgement mode used
420/// * `scrape_url` - URL that was scraped
421///
422/// # Returns
423/// * `Result<(), Box<dyn std::error::Error>>` - Success or an error
424fn save_results(
425    proxies: &[Proxy],
426    source: &Source,
427    filestore: &Filestore,
428    dry: bool,
429    raw_response: Option<String>,
430    mode: JudgementMode,
431    scrape_url: &str,
432) -> Result<(), Box<dyn std::error::Error>> {
433    if dry {
434        return Ok(());
435    }
436
437    // Save proxies if they exist
438    if !proxies.is_empty() {
439        filestore.save_proxies(proxies, "proxies")?;
440    }
441
442    // Save raw response data if in full mode
443    if mode == JudgementMode::Full && raw_response.is_some() {
444        let timestamp = chrono::Utc::now().format("%Y%m%d-%H%M%S");
445        let sanitized_url = utils::sanitize_url_for_filename(scrape_url);
446        let dump_filename = format!("{timestamp}-{sanitized_url}.txt");
447
448        if let Err(e) = std::fs::write(dump_filename.clone(), raw_response.unwrap()) {
449            eprintln!("Failed to save raw response: {e}");
450        } else {
451            println!("Raw response saved to {dump_filename}");
452        }
453    }
454
455    // Load existing sources
456    let mut sources = filestore.load_sources("sources").unwrap_or_default();
457
458    // Update or add new source
459    if let Some(pos) = sources.iter().position(|s| s.url == source.url) {
460        sources[pos] = source.clone();
461    } else {
462        sources.push(source.clone());
463    }
464
465    filestore.save_sources(&sources, "sources")?;
466    println!("Source saved successfully");
467
468    Ok(())
469}

471/// Scrapes and processes proxies from a source URL.
472///
473/// This function handles the entire proxy scraping workflow:
474/// 1. Loads configuration and initializes components
475/// 2. Scrapes proxies from the specified URL
476/// 3. Tests and enriches the proxies based on the judgement mode
477/// 4. Saves the results to the filestore
478///
479/// # Arguments
480///
481/// * `scrape` - URL to scrape for proxies
482/// * `config` - Path to configuration folder (default: 'data')
483/// * `useragent` - Custom User-Agent string to use for requests
484/// * `pattern` - Custom regex pattern for finding proxies
485/// * `judge` - Judgement mode determining test intensity:
486///   - None (0): No testing, just scrape
487///   - Quick (1): Basic connectivity testing
488///   - Full (2): Comprehensive testing with metadata collection
489/// * `dry` - If true, don't save results to persistent storage
490///
491/// # Returns
492///
493/// * `()` - The function exits the process with an appropriate status code
494async fn handle_source_command(
495    scrape: String,
496    config: Option<String>,
497    useragent: Option<String>,
498    pattern: Option<String>,
499    judge: JudgementMode,
500    dry: bool,
501) {
502    // Load configuration
503    let config_path = config.unwrap_or_else(|| "data".to_string());
504    let filestore = match setup_filestore(&config_path) {
505        Ok(fs) => fs,
506        Err(e) => {
507            eprintln!("Failed to initialize filestore: {e}");
508            std::process::exit(1);
509        }
510    };
511
512    // Initialize source with provided options
513    let source = match Source::new(
514        scrape.clone(),
515        useragent.unwrap_or_else(|| utils::get_random_user_agent().to_string()),
516        pattern.unwrap_or_else(|| defaults::regex_patterns::IP_PORT.to_string()),
517    ) {
518        Ok(s) => s,
519        Err(e) => {
520            eprintln!("Failed to create source: {e}");
521            std::process::exit(1);
522        }
523    };
524
525    // Create requestor for fetching
526    let requestor = match Requestor::new() {
527        Ok(r) => r,
528        Err(e) => {
529            eprintln!("Failed to create requestor: {e}");
530            std::process::exit(1);
531        }
532    };
533
534    // Fetch proxies from the source
535    println!("Scraping proxies from {scrape}");
536    let (proxies, raw_response) = match source.fetch_proxies_with_response(&requestor).await {
537        Ok((proxies, response)) => (proxies, response),
538        Err(e) => {
539            eprintln!("Failed to fetch proxies: {e}");
540            std::process::exit(1);
541        }
542    };
543
544    println!("Found {} proxies", proxies.len());
545
546    // Test and enrich proxies if requested
547    let proxies = match test_and_enrich_proxies(proxies, judge).await {
548        Ok(proxies) => {
549            if judge != JudgementMode::None {
550                // Count working proxies
551                let working = proxies
552                    .iter()
553                    .filter(|p| p.check_success_rate() > 0)
554                    .count();
555                println!("\nWorking proxies: {}/{}", working, proxies.len());
556            }
557            proxies
558        }
559        Err(e) => {
560            eprintln!("Failed during proxy testing: {e}");
561            std::process::exit(1);
562        }
563    };
564
565    // Save results
566    let raw_response_to_save = if judge == JudgementMode::Full {
567        Some(raw_response)
568    } else {
569        None
570    };
571
572    if let Err(e) = save_results(
573        &proxies,
574        &source,
575        &filestore,
576        dry,
577        raw_response_to_save,
578        judge,
579        &scrape,
580    ) {
581        eprintln!("Failed to save results: {e}");
582        std::process::exit(1);
583    }
584
585    std::process::exit(0);
586}

588/// Parses a proxy URL string into a Proxy object.
589///
590/// # Arguments
591/// * `url` - The proxy URL in format protocol://ip:port
592///
593/// # Returns
594/// * `Result<Proxy, String>` - The parsed Proxy object or an error message
595fn parse_proxy_url(url: &str) -> Result<Proxy, String> {
596    // Basic URL parsing - protocol://ip:port
597    let parts: Vec<&str> = url.split("://").collect();
598    if parts.len() != 2 {
599        return Err("Invalid proxy URL format. Expected: protocol://ip:port".to_string());
600    }
601
602    let lower = if parts.is_empty() {
603        return Err("No protocol specified in proxy URL".to_string());
604    } else {
605        parts[0].to_lowercase()
606    };
607
608    let protocol = match lower.as_str() {
609        "http" => ProxyType::Http,
610        "https" => ProxyType::Https,
611        "socks4" => ProxyType::Socks4,
612        "socks5" => ProxyType::Socks5,
613        _ => return Err("Invalid protocol. Use: http, https, socks4, or socks5".to_string()),
614    };
615
616    let addr_parts: Vec<&str> = parts[1].split(':').collect();
617    if addr_parts.len() != 2 {
618        return Err("Invalid address format. Expected: ip:port".to_string());
619    }
620
621    let Ok(ip) = IpAddr::from_str(addr_parts[0]) else {
622        return Err("Invalid IP address".to_string());
623    };
624
625    let Ok(port) = addr_parts[1].parse::<u16>() else {
626        return Err("Invalid port number".to_string());
627    };
628
629    Ok(Proxy::new(protocol, ip, port, AnonymityLevel::Anonymous))
630}

652/// Main function that handles CLI argument parsing and command dispatching.
653/// Uses the clap crate for command-line argument parsing.
654#[tokio::main]
655async fn main() {
656    // Helper function to convert our LogLevel enum to log::LevelFilter
657    fn log_level_to_filter(log_level: LogLevel) -> log::LevelFilter {
658        match log_level {
659            LogLevel::Error => log::LevelFilter::Error,
660            LogLevel::Warn => log::LevelFilter::Warn,
661            LogLevel::Info => log::LevelFilter::Info,
662            LogLevel::Debug => log::LevelFilter::Debug,
663            LogLevel::Trace => log::LevelFilter::Trace,
664        }
665    }
666
667    let cli = Cli::parse();
668
669    // Set up logging based on log level
670    // Convert LogLevel to log::LevelFilter
671    let level_filter = log_level_to_filter(cli.log_level);
672    pretty_env_logger::formatted_builder()
673        .filter_level(level_filter)
674        .init();
675
676    // Process command and arguments
677    match cli.command {
678        None => {
679            Cli::command().print_help().unwrap();
680            std::process::exit(1);
681        }
682        Some(Commands::Config { create, validate }) => {
683            handle_config_command(create, validate);
684        }
685        Some(Commands::Proxy { judge, dry }) => {
686            handle_proxy_command(judge, dry).await;
687        }
688        Some(Commands::Source {
689            scrape,
690            config,
691            useragent,
692            pattern,
693            judge,
694            dry,
695        }) => {
696            handle_source_command(scrape, config, useragent, pattern, judge, dry).await;
697        }
698    }
699}
